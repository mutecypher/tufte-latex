\nomenclature{$G=(V,E)$}{graph with vertex set $V$ and edge set $E$}

\nomenclature{$N(v)$}{neighborhood of $v$, i.e. $N(v) = \{ w \in V\ |\ \{v,w\} \in E\}$}
\nomenclature{$N[v]$}{closed neighborhood of $v$, i.e. $N[v] = N(v) \cup \{v\}$}
\nomenclature{$E(G)$}{edge set of a graph $G$}
\nomenclature{$G(V')$}{subgraph of a graph $G=(V,E)$ induced by a subset of vertices $V'\subseteq V$}
